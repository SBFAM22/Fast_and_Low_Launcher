    public class ModEntry : EntryMod
    {
        public static GameObject ModdedGameObject { get; set; }

        public override void LoadMod()
        {
            int random = new System.Random().Next(0, 100);
            ModdedGameObject = new GameObject("DummyObject" + random.ToString()).AddComponent<Mod>().gameObject;
            base.LoadMod();
        }
    }

    }
